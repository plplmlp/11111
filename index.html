// Cloudflare Pages Function - 图片生成 export async function onRequestPost(context) { const { request, env } = context; const headers = { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' }; try { const body = await request.json(); const { productName, sellingPoints, types, refDescription, refImage } = body; if (!productName || !types || types.length === 0) { return new Response(JSON.stringify({ error: 'Missing required fields' }), { status: 400, headers }); } const type = types[0]; const prompt = buildPrompt(type, productName, sellingPoints, refDescription); const apiKey = env.SILICONFLOW_API_KEY; if (!apiKey) { return new Response(JSON.stringify({ error: 'API Key not configured' }), { status: 500, headers }); } const postData = { model: 'Kwai-Kolors/Kolors', prompt: prompt, negative_prompt: 'NO person, NO man, NO woman, NO model, product only, no face, no body, no cartoon, not anime', image_size: '1024x1024', num_inference_steps: 20, guidance_scale: 7.5, n: 1 }; // 如果有参考图，添加 img2img if (refImage) { postData.image = refImage; } const response = await fetch('https://api.siliconflow.cn/v1/images/generations', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${apiKey}` }, body: JSON.stringify(postData) }); const data = await response.json(); if (data.images && data.images[0] && data.images[0].url) { return new Response(JSON.stringify({ success: true, images: { [type]: data.images[0].url } }), { headers }); } else if (data.error) { return new Response(JSON.stringify({ error: data.error.message || 'API Error' }), { status: 500, headers }); } else { return new Response(JSON.stringify({ error: 'Unknown response from API' }), { status: 500, headers }); } } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers }); } } export async function onRequestOptions(context) { return new Response(null, { status: 200, headers: { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type' } }); } function buildPrompt(type, productName, sellingPoints, refDescription) { const basePrompt = `Professional e-commerce product photography of ${productName}`; const sellingText = sellingPoints ? `, ${sellingPoints}` : ''; const typePrompts = { 'whitebg': `${basePrompt}${sellingText}, pure white background (#FFFFFF), studio lighting, product centered occupying 80% of frame, sharp focus, commercial catalog quality, photorealistic, 8K quality`, 'model': `Fashion photography, handsome young Asian male model age 25, fit build, wearing ${productName}${sellingText}, natural relaxed pose, light gray studio background, professional studio lighting, sharp focus on clothing, editorial fashion photo, photorealistic skin texture, 8K quality`, 'scene': `Lifestyle photography, stylish young Asian man wearing ${productName}${sellingText}, sitting in modern minimalist cafe with large windows, natural sunlight, warm golden hour lighting, shallow depth of field, bokeh background, relaxed candid pose, editorial fashion photo, photorealistic, 8K quality` }; return typePrompts[type] || basePrompt; }
